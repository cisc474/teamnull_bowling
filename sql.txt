CREATE TABLE customer(customerID INT AUTO_INCREMENT, firstName VARCHAR(25), lastName VARCHAR(25), street VARCHAR(25), city VARCHAR(25), state VARCHAR(25), zip VARCHAR(9), homePhone VARCHAR(10), cellPhone VARCHAR(10), email VARCHAR(30), PRIMARY KEY(customerID)) ENGINE="INNODB";

CREATE TABLE user(userName VARCHAR(25), password VARCHAR(25), PRIMARY KEY(userName)) ENGINE="INNODB";

CREATE TABLE ball(ballID INT AUTO_INCREMENT, 
customerID INT, 
date DATE,
hand VARCHAR(10), 
layout VARCHAR(25),
BHPosition VARCHAR(15), 
BHSize VARCHAR(15),
BHDepth VARCHAR(15),
conv VARCHAR(10),
FT VARCHAR(10),
weight REAL
pin VARCHAR(10),
surface VARCHAR(15),
size VARCHAR(10),
depth VARCHAR(10),
PAP_Horz VARCHAR(10),
PAP_Vert VARCHAR(10),
CLT VARCHAR(25),
thumb VARCHAR(15),
fingers VARCHAR(25),
price REAL, 
clerk VARCHAR(40),
LVertPitch VARCHAR(5),
LHorzPitch VARCHAR(5),
LPitchValue VARCHAR(7),
RVertPitch VARCHAR(5),
RHorzPitch VARCHAR(5),
RPitchValue VARCHAR(7),
TVertPitch VARCHAR(5),
THorzPitch VARCHAR(5),
LDecimal REAL, 
RDecimal REAL, 
LDiameter VARCHAR(7),
LComment VARCHAR(300),
RDiameter VARCHAR(7),
RComment VARCHAR(300),
TDiameter VARCHAR(7),
TComment VARCHAR(300),
FFDIST VARCHAR(7),
LFTDIST VARCHAR(7),
RFTDIST VARCHAR(7),
OvalCutLeft VARCHAR(7), OvalCutRight VARCHAR(7),
LCC BOOLEAN, RCC BOOLEAN, LFS BOOLEAN, RFS BOOLEAN, 
comments VARCHAR(300), PRIMARY KEY(ballID), FOREIGN KEY(customerID) REFERENCES customer(customerID) ON DELETE CASCADE ON UPDATE CASCADE) ENGINE = "INNODB";
